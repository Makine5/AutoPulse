  async function sendChat() {
    const input = document.getElementById('chat-input');
    const message = input.value.trim();
    if (!message) return;
    if (!chatBusinessInfo) {
      chatBusinessInfo = document.getElementById('chatbot-business-info').value.trim() || 'A helpful business assistant.';
    }
    const messages = document.getElementById('chat-messages');
    messages.innerHTML += `
${message}
`;
    input.value = '';
    const typingId = 'typing-' + Date.now();
    messages.innerHTML += `
Typing...
`;
    messages.scrollTop = messages.scrollHeight;
    try {
      const res = await fetch('/chatbot', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ message, businessInfo: chatBusinessInfo, history: chatHistory })
      });
      if (!res.ok) throw new Error('Server error ' + res.status);
      const data = await res.json();
      const reply = data.reply || 'Sorry, I could not process that.';
      chatHistory.push({ role: 'user', content: message });
      chatHistory.push({ role: 'assistant', content: reply });
      document.getElementById(typingId)?.remove();
      messages.innerHTML += `
${reply}
`;
    } catch(e) {
      document.getElementById(typingId)?.remove();
      messages.innerHTML += `
Sorry, something went wrong: ${e.message}
`;
    }
    messages.scrollTop = messages.scrollHeight;
  }

  async function scanEmails() {
    const btn = document.getElementById('scan-btn');
    btn.disabled = true;
    btn.textContent = '⏳ Scanning...';

    try {
      const res = await fetch('/scan-emails', { credentials: 'include' });
      const data = await res.json();

      if (data.error) {
        alert('Error: ' + data.error);
        btn.disabled = false;
        btn.textContent = '⚡ SCAN MY INBOX NOW';
        return;
      }

      // Update stats
      const prev = {
        scanned: parseInt(document.getElementById('stat-scanned').textContent) || 0,
        deleted: parseInt(document.getElementById('stat-deleted').textContent) || 0,
        replied: parseInt(document.getElementById('stat-replied').textContent) || 0,
      };
      document.getElementById('stat-scanned').textContent = prev.scanned + data.scanned;
      document.getElementById('stat-deleted').textContent = prev.deleted + data.deleted;
      document.getElementById('stat-replied').textContent = prev.replied + data.replied;

      // Update log
      const log = document.getElementById('activity-log');
      if (data.log && data.log.length > 0) {
        // Save deleted ones to history
        data.log.forEach(item => {
          if (item.action.includes('Deleted')) {
            saveDeletedEmail(item.subject, item.from);
          }
        });
        const items = data.log.map(item => `
          

            

              
${item.subject}

              
${item.from}

              ${item.reply ? `
💬 Replied: "${item.reply.substring(0, 120)}${item.reply.length > 120 ? '...' : ''}"
` : ''}
            

            
${item.action}

          

        `).join('');
        log.innerHTML = items + log.innerHTML;
      } else {
        log.innerHTML = '
No new emails found.
';
      }

    } catch(e) {
      alert('Scan failed. Make sure your Gmail is connected.');
    }

    btn.disabled = false;
    btn.textContent = '⚡ SCAN MY INBOX NOW';
  }

  function switchLogTab(tab) {
    document.querySelectorAll('.log-tab').forEach(b => b.classList.remove('active'));
    event.target.classList.add('active');
    document.getElementById('log-session').style.display = tab === 'session' ? 'block' : 'none';
    document.getElementById('log-deleted').style.display = tab === 'deleted' ? 'block' : 'none';
    if (tab === 'deleted') renderDeletedHistory();
  }

  function getDeletedHistory() {
    try { return JSON.parse(localStorage.getItem('ap_deleted') || '[]'); } catch(e){ return []; }
  }

  function saveDeletedEmail(subject, from) {
    const history = getDeletedHistory();
    history.unshift({ subject, from, date: new Date().toLocaleString() });
    if (history.length > 200) history.splice(200);
    localStorage.setItem('ap_deleted', JSON.stringify(history));
  }

  function clearDeletedHistory() {
    if (confirm('Clear all deleted email history?')) {
      localStorage.removeItem('ap_deleted');
      renderDeletedHistory();
    }
  }

  function renderDeletedHistory() {
    const history = getDeletedHistory();
    const el = document.getElementById('deleted-log');
    if (history.length === 0) {
      el.innerHTML = '
No deleted emails yet. Run a scan to start!
';
      return;
    }
    el.innerHTML = history.map(item => `
      

        

          
${item.subject}

          
${item.from}

          
${item.date}

        

        
🗑️ Deleted (scam)

      

    `).join('');
  }
